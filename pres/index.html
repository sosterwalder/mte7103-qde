Rendering: Vertex shader

 

// VERTEX SHADER

#version 330

// in / attributes
attribute vec4 a_position;

// uniforms
uniform mat4 u_mvpMatrix;

// Main method of the shader.
void main()
{
    // gl_Position contains the position of the current vertex.
    // a_position are the specified vertices, which is in this
    // case a simple rectangle filling the whole screen.
    gl_Position = u_mvpMatrix * a_position;
}
